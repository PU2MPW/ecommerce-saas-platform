import { createClient } from '@/lib/supabase/client';\
import type { AddressFormData, ShippingOption } from '@/types/checkout';\
import type { CartItem } from '@/hooks/useCart';\
\
export interface CreateOrderInput {\
  tenantId: string; userId: string; items: CartItem[];\
  address: AddressFormData; shipping: ShippingOption; subtotal: number; discount?: number;\
}\
\
export interface CreateOrderOutput { orderId: string; total: number; status: 'pending'; }\
\
export async function createOrder(input: CreateOrderInput): Promise<CreateOrderOutput> {\
  const supabase = createClient();\
  const { data: order, error } = await supabase.from('orders').insert({\    tenant_ id: input.tenantId, user_id: input.userId, status: 'pending',\
    subtotal: input.subtotal, shipping_cost: input.shipping.price, total: input.subtotal + input.shipping.price - (input.discount || 0),\
    shipping_ address: input.address, shipping_service: input.shipping.service, shipping_deadline: input.shipping.deadline,\
  }).select('id').single();\
  if (error || !order) throw new Error('Erro ao criar pedido');\
  const { error: itemsErr } = await supabase.from('order_items').insert(\
    input.items.map(item => ({ order_id: order.id, product_id: item.productId, quantity: item.quantity, price: item.price })));\
  if (itemsErr) { await supabase.from('orders').delete().eq('id', order.id); throw new Error('Erro ao criar itens'); }\
  return { orderId: order.id, total: input.subtotal + input.shipping.price, status: 'pending' };\
}\

' use client';\
import { useState, useCallback } from 'react';\
import { createOrder } from '@/lib/orders';\
import { useRouter } from 'next/navigation';\
\
export function useCheckout() {\
  const [loading, setLoading] = useState(false);\
  const [error, setError] = useState<string | null>(null);\
  const router = useRouter();\
\
  const submit = useCallback(async (input: Parameters<typeof createOrder>[0]) => {\
    setLoading(true); setError(null);\
    try { const result = await createOrder(input); return result; }\
    catch (err) { setError(err instanceof Error ? err.message : 'Erro'); throw err; }\
    finally { setLoading(false); }\
  }, []);\
\
  const goToSuccess = useCallback((orderId: string) => { router.push(`/checkout/success?orderId=${orderId}`); }, [router]);\
\
  return { loading, error, submit, goToSuccess };\
}\

import type { CartItem } from '@/hooks/useCart';\
\
interface OrderReviewProps { items: CartItem[]; address: AddressFormData; shipping: ShippingOption; }\
export function OrderReview({ items, address, shipping }: OrderReviewProps) {\
  const { loading, error, submit, goToSuccess } = useCheckout();\
  const subtotal = items.reduce((sum, item) => sum + item.price * item.quantity, 0);\
  const total = subtotal + shipping.price;\
