```typescript
'use client';
import { useState, useCallback } from 'react';

interface SavedAddress {
  id: string; label: string; zipcode: string; street: string;
  number: string; complement?: string; neighborhood: string;
  city: string; state: string; isDefault: boolean;
}

export function useAddress() {
  const [addresses, setAddresses] = useState<SavedAddress[]>([]);
  
  const fetchAddresses = useCallback(async (userId: string) => {
    const res = await fetch(`/api/addresses?userId=${userId}`);
    if (res.ok) {
      const data = await res.json();
      setAddresses(data);
    }
  }, []);
  
  const createAddress = useCallback(async (address: Omit<SavedAddress, 'id'>) => {
    const res = await fetch('/api/addresses', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(address),
    });
    if (res.ok) {
      const created = await res.json();
      setAddresses(prev => [...prev, created]);
      return created;
    }
    throw new Error('Erro ao salvar endereço');
  }, []);
  
  return { addresses, fetchAddresses, createAddress };
}
```

1. **src/hooks/useShipping.ts** — Hook para cálculo de frete:
```typescript
'use client';
import { useState, useCallback } from 'react';
import type { ShippingCalcRequest, ShippingOption } from '@/types/checkout';

export function useShipping() {
  const [options, setOptions] = useState<ShippingOption[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  
  const calculate = useCallback(async (req: ShippingCalcRequest) => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch('/api/shipping/calculate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(req),
      });
      
      if (!res.ok) throw new Error('Erro ao calcular frete');
      
      const data = await res.json();
      setOptions(data.options || []);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Erro desconhecido');
      setOptions([]);
    } finally {
      setLoading(false);
    }
  }, []);
  
  return { options, loading, error, calculate };
}
```

```typescript
'use client';
import { useEffect } from 'react';
import { useShipping } from '@/hooks/useShipping';
import type { ShippingOption, ShippingCalcRequest } from '@/types/checkout';
import { useCart } from '@/hooks/useCart';

interface ShippingOptionsProps {
  zipcode: string;
  selectedId?: string;
  onSelect: (option: ShippingOption) => void;
}

export function ShippingOptions({ zipcode, selectedId, onSelect }: ShippingOptionsProps) {
  const { items } = useCart();
  const { options, loading, error, calculate } = useShipping();
  
  // Recalcula quando CEP muda
  useEffect(() => {
    if (zipcode && items.length > 0) {
      // Obter dimensões dos produtos do carrinho
      const req: ShippingCalcRequest = {
        zipcode,
        items: items.map(item => ({
          weight: 0.5, // TODO: buscar do produto
          height: 20,
          width: 15,
          depth: 10,
          quantity: item.quantity,
        })),
      };
      calculate(req);
    }
  }, [zipcode, items.length]);
  
  if (loading) {
    return (
      <div className="flex items-center justify-center py-8">
        <div className="h-6 w-6 animate-spin rounded-full border-2 border-primary border-t-transparent" />
        <span className="ml-2 text-gray-500">Calculando fretes...</span>
      </div>
    );
  }
  
  if (error) {
    return (
      <div className="rounded-md bg-red-50 p-4">
        <p className="text-sm text-red-600">{error}</p>
      </div>
    );
  }
  
  if (options.length === 0) {
    return (
      <div className="rounded-md bg-yellow-50 p-4 text-sm text-yellow-700">
        Informe um CEP válido para ver as opções de entrega.
      </div>
    );
  }
  
  return (
    <div className="space-y-2">
      {options.map(option => (
        <label
          key={option.id}
          className={`flex items-center justify-between rounded-lg border p-4 cursor-pointer transition-colors ${
            selectedId === option.id
              ? 'border-primary bg-primary/5'
              : 'border-gray-200 hover:border-gray-300'
          }`}
        >
          <div className="flex items-center gap-3">
            <input
              type="radio"
              name="shipping"
              value={option.id}
              checked={selectedId === option.id}
              onChange={() => onSelect(option)}
              className="h-4 w-4 text-primary"
            />
            <div>
              <p className="font-medium">{option.label}</p>
              <p className="text-sm text-gray-500">Entrega em até {option.deadline} dias úteis</p>
            </div>
          </div>
          <p className="font-semibold">
            {new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(option.price / 100)}
          </p>
        </label>
      ))}
    </div>
  );
}
```
