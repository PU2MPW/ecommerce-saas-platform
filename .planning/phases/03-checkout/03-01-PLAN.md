```typescript
// API pública dos Correios para cálculo de frete
// https://www.correios.com.br/a-a-z/calculador-de-precos-e-prazos

interface CorreiosInput {
  sCepOrigem: string;    // CEP origem (sem hífen)
  sCepDestino: string;   // CEP destino (sem hífen)
  nVlPeso: string;       // Peso em kg
  nVlAltura: string;    // Altura em cm
  nVlLargura: string;    // Largura em cm
  nVlComprimento: string; // Comprimento em cm
  nCdServico: string;    // 40010=SEDEX, 41106=PAC
}

interface CorreiosOutput {
  Servicos: Array<{
    Codigo: string;
    Valor: string;       // "R$ 12,34"
    PrazoEntrega: string; // "5"
  }>;
}

export async function calculateCorreios(input: CorreiosInput): Promise<CorreiosOutput> {
  const params = new URLSearchParams({
    ...input,
    StrRetorno: 'xml',
  });
  
  const response = await fetch(
    `https://correios.com.br/api/v2/calculate-price?${params}`
  );
  
  if (!response.ok) {
    throw new Error('Correios API error');
  }
  
  return response.json();
}
```

**src/lib/shipping/calculator.ts** — Orquestra cálculo:
```typescript
import { calculateCorreios } from './correios';
import type { ShippingCalcRequest, ShippingOption } from '@/types/checkout';

const SHIPPING_SERVICES = [
  { carrier: 'Correios', service: 'PAC', code: '41106', label: 'PAC' },
  { carrier: 'Correios', service: 'SEDEX', code: '40010', label: 'SEDEX' },
  { carrier: 'Correios', service: 'SEDEX10', code: '40010', label: 'SEDEX Hoje' },
];

export async function calculateShipping(
  req: ShippingCalcRequest,
  tenantConfig: { zipcode: string }
): Promise<ShippingOption[]> {
  // Normaliza CEP origem (remove hífen)
  const originZipcode = tenantConfig.zipcode.replace(/\D/g, '');
  const destZipcode = req.zipcode.replace(/\D/g, '');
  
  const options: ShippingOption[] = [];
  
  for (const svc of SHIPPING_SERVICES) {
    try {
      const result = await calculateCorreios({
        sCepOrigem: originZipcode,
        sCepDestino: destZipcode,
        nVlPeso: String(req.items.reduce((sum, i) => sum + i.weight * i.quantity, 0)),
        nVlAltura: String(Math.max(...req.items.map(i => i.height))),
        nVlLargura: String(Math.max(...req.items.map(i => i.width))),
        nVlComprimento: String(Math.max(...req.items.map(i => i.depth))),
        nCdServico: svc.code,
      });
      
      if (result?.Servicos?.[0]) {
        const price = parseFloat(result.Servicos[0].Valor.replace(/[^\d,]/g, '').replace(',', '.'));
        const deadline = parseInt(result.Servicos[0].PrazoEntrega) || 0;
        
        options.push({
          id: `${svc.carrier}-${svc.service}`.toLowerCase(),
          carrier: svc.carrier,
          service: svc.service,
          price: Math.round(price * 100),
          deadline,
          label: `${svc.label} - ${deadline} dias úteis`,
          deliveryDate: new Date(Date.now() + deadline * 86400000).toISOString(),
        });
      }
    } catch (err) {
      // Log error, continue to next service
      console.error(`Failed ${svc.service}:`, err);
    }
  }
  
  return options.sort((a, b) => a.price - b.price);
}
```

**src/app/api/shipping/calculate/route.ts** — Endpoint:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { calculateShipping } from '@/lib/shipping/calculator';
import type { ShippingCalcRequest } from '@/types/checkout';

export async function POST(req: NextRequest) {
  try {
    const body: ShippingCalcRequest = await req.json();
    
    // Validação de CEP
    const cepRegex = /^\d{5}-?\d{3}$/;
    if (!cepRegex.test(body.zipcode)) {
      return NextResponse.json(
        { error: 'CEP inválido. Formato esperado: 00000-000' },
        { status: 400 }
      );
    }
    
    // Obter CEP origem do tenant (de tenant_ettings)
    const tenantZipcode = '00000000'; // TODO: buscar do tenant settings
    
    const options = await calculateShipping(body, { zipcode: tenantZipcode });
    
    return NextResponse.json({
      options,
      zipcode: body.zipcode,
      calculatedAt: new Date().toISOString(),
    });
  } catch (err) {
    console.error('Shipping calculation error:', err);
    return NextResponse.json(
      { error: 'Erro ao calcular frete. Tente novamente.' },
      { status: 500 }
    );
  }
}
```
