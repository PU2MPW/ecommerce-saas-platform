import { createClient } from '@/lib/supabase/client';
import type { AddressFormData, ShippingOption } from '@/types/checkout';
import type { CartItem } from '@/hooks/useCart';

export interface CreateOrderInput {
  tenantId: string; userId: string;
  items: CartItem[]; address: AddressFormData; shipping: ShippingOption;
  subtotal: number; discount?: number;
}

export interface CreateOrderOutput {
  orderId: string; total: number; status: 'pending';
}

export async function createOrder(input: CreateOrderInput): Promise<CreateOrderOutput> {
  const supabase = createClient();
  const shippingCost = input.shipping.price;
  const total = input.subtotal + shippingCost - (input.discount || 0);

  // 1. Insert order header
  const { data: order, error: orderError } = await supabase
    .from('orders')
    .insert({
      tenant_id: input.tenantId, user_id: input.userId,
      status: 'pending', subtotal: input.subtotal, shipping_cost: shippingCost, total,
      shipping_address: input.address, shipping_service: input.shipping.service,
      shipping_deadline: input.shipping.deadline,
    })
    .select('id').single();

  if (orderError || !order) throw new Error('Erro ao criar pedido: ' + (orderError?.message || ''));

  // 2. Insert order items
  const items = input.items.map(item => ({
    order_id: order.id, product_id: item.productId,
    quantity: item.quantity, price: item.price,
  }));

  const { error: itemsError } = await supabase.from('order_items').insert(items);
  if (itemsError) {
    await supabase.from('orders').delete().eq('id', order.id);
    throw new Error('Erro ao criar itens: ' + itemsError.message);
  }

  return { orderId: order.id, total, status: 'pending' };
}

' use client';
import { useState, useCallback } from 'react';
import { createOrder, type CreateOrderInput } from '@/lib/orders';
import { useRouter } from 'next/navigation';

export function useCheckout() {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [orderId, setOrderId] = useState<string | null>(null);
  const router = useRouter();

  const submit = useCallback(async (input: CreateOrderInput) => {
    setLoading(true); setError(null);
    try {
      const result = await createOrder(input);
      setOrderId(result.orderId);
      return result;
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Erro');
      throw err;
    } finally { setLoading(false); }
  }, []);

  const goToSuccess = useCallback((id: string) => {
    router.push(`/checkout/success?orderId=${id}`);
  }, [router]);

  return { loading, error, orderId, submit, goToSuccess };
}

' use client';
import { useCheckout } from '@/hooks/useCheckout';
import type { AddressFormData, ShippingOption } from '@/types/checkout';
import type { CartItem } from '@/hooks/useCart';
interface OrderReviewProps {
  items: CartItem[]; address: AddressFormData; shipping: ShippingOption;
}
export function OrderReview({ items, address, shipping }: OrderReviewProps) {
  const { loading, error, submit, goToSuccess } = useCheckout();
  const subtotal = items.reduce((sum, item) => sum + item.price * item.quantity, 0);
  const total = subtotal + shipping.price;

  const handleConfirm = async () => {
    const result = await submit({ tenantId: 'tenant-id', userId: 'user-id', items, address, shipping, subtotal });
    goToSuccess(result.orderId);
  };

  return (
    <div className="rounded-lg border border-gray-200 p-6">
      <h2 className="text-xl font-semibold mb-6">Revise seu Pedido</h2>
      <div className="mb-6"> <h3 className="text-lg font-medium mb-2">Endereco</h3>
        <div className="rounded-md bg-gray-50 p-4 text-sm">
          <p className="font-medium">{address.street}, {address.number}</p>
          <p>{address.neighborhood} - {address.city}/{address.state}</p>
          <p>CEP: {address.zipcode}</p> </div> </div>
      <div className="mb-6"> <h3 className="text-lg font-medium mb-2">Entrega</h3>
        <div className="rounded-md bg-gray-50 p-4 text-sm flex justify-between">
          <div><p className="font-medium">{shipping.label}</p><p className="text-gray-500">{shipping.deadline} dias</p></div>
          <p className="font-medium">{new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(shipping.price / 100)}</p> </div> </div>
      <div className="mb-6"> <h3 className="text-lg font-medium mb-2">Itens ({items.length})</h3>
        <div className="rounded-md border border-gray-200 divide-y divide-gray-200">
          {items.map(item => (
            <div key={item.productId} className="flex gap-4 p-4">
              <img src={item.imageUrl} alt={item.name} className="h-16 w-16 rounded object-cover" />
              <div className="flex-1"><p className="font-medium">{item.name}</p><p className="text-sm text-gray-500">Qty: {item.quantity}</p></div>
              <p className="font-medium">{new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(item.price * item.quantity / 100)}</p> </div> ))} </div> </div>
      <div className="border-t pt-4 space-y-2">
        <div className="flex justify-between text-sm"><span>Subtotal</span><span>{new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(subtotal / 100)}</span></div>
        <div className="flex justify-between text-sm"><span>Frete</span><span>{new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(shipping.price / 100)}</span></div>
        <div className="flex justify-between text-xl font-semibold pt-2 border-t">
          <span>Total</span><span className="text-primary">{new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' }).format(total / 100)}</span> </div> </div>
      {error && <div className="mt-4 rounded-md bg-red-50 p-4 text-sm text-red-600">{error}</div>}
      <div className="mt-6 flex gap-4">
        <button type="button" onClick={() => window.history.back()}
